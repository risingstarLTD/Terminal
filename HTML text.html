            function getRecommendation(predictions) {

                if (useLiveAnalyst && trainedModel) {

                    const aiPredictionId = getAIPrediction(predictions);

                    if(aiPredictionId) {

                        const recommendation = predictions.find(p => p.id === aiPredictionId);

                        return { type: 'play', recommendations: [recommendation], reason: 'Contextual AI Engine Recommendation', confidence: 'AI' };

                    }

                }


                const stats = getStats();

                const lastWin = confirmedWinsLog.length > 0 ? confirmedWinsLog[0] : null;


                let scoredPredictions = predictions.map(p => {

                    const pStats = stats[p.id] || { hits: 0, total: 0, streak: 0, hitRate: 0 };

                    let score = pStats.hitRate;


                    if (useWeightedZone) {

                        const neighbourScores = getNeighbourScores();

                        const zoneScore = p.hitZone.reduce((acc, num) => acc + (neighbourScores[num] || 0), 0);

                        score += zoneScore / weightedZoneDivisor;

                    }

                    if (useProximityBoost && lastWin) {

                        const distance = getShortestPocketDistance(lastWin.winningNumber, p.hitZone);

                        if (distance !== -1) score += (1 / (distance + 1)) * proximityBoostMultiplier;

                    }

                    if (useDueForHit) {

                        const expectedRate = 1 / activePredictionTypes.length;

                        const performanceDiff = expectedRate - pStats.hitRate;

                        if (performanceDiff > 0) score += performanceDiff * dueForHitMultiplier;

                    }

                    if(useCorrelationAnalysis && lastWin && lastWin.successfulPrediction) {

                        const lastWinningGroup = lastWin.successfulPrediction.id;

                        if(correlationMatrix[lastWinningGroup] && correlationMatrix[lastWinningGroup][p.id]) {

                            score += correlationMatrix[lastWinningGroup][p.id];

                        }

                    }

                    return { ...p, score, stats: pStats };

                });

                

                scoredPredictions.sort((a, b) => b.score - a.score);


                if (useLowestPocketDistance && lastWin) {

                    const closest = predictions.map(p => ({...p, distance: getShortestPocketDistance(lastWin.winningNumber, p.hitZone)}))

                                               .filter(p => p.distance !== -1).sort((a,b) => a.distance - b.distance);

                    if(closest.length > 0 && closest[0].distance <= 1) {

                        scoredPredictions = scoredPredictions.map(p => ({...p, score: p.id === closest[0].id ? p.score + lowestPocketDistanceMultiplier : p.score }))

                                                             .sort((a, b) => b.score - a.score);

                    }

                }


                let finalRecommendations = [scoredPredictions[0]];

                

                if(useMultiGroupBetting && scoredPredictions[0].score > 0) {

                    const topScore = scoredPredictions[0].score;

                    finalRecommendations = scoredPredictions.filter(p => p.score >= topScore * multiBettingScoreThreshold);

                }


                let reason = "Top score based on active strategies.";

                let confidence = "Standard";


                const topRec = finalRecommendations[0];

                const highHitRate = topRec.stats.hitRate > 0.6;

                const longStreak = topRec.stats.streak >= 3;

                const trendConfirmed = useTrendConfirmation && lastWin && lastWin.successfulPrediction && lastWin.successfulPrediction.id === topRec.id;


                if (useLessStrict) {

                    if (highHitRate || longStreak) confidence = "High Confidence";

                } else {

                    if ((highHitRate || longStreak) && trendConfirmed) confidence = "High Confidence";

                }

                if (useTrendConfirmation && lastWin && lastWin.successfulPrediction) {

                    if (lastWin.successfulPrediction.id !== topRec.id) {

                        return { type: 'wait', reason: `Waiting for ${lastWin.successfulPrediction.label} trend. Current top is ${topRec.label}.`};

                    }

                }

                if (usePatternRecognition && activePatterns.length > 0) {

                    const patternRec = activePatterns[0].next;

                    const patternPrediction = scoredPredictions.find(p => p.id === patternRec);

                    if(patternPrediction) {

                        finalRecommendations = [patternPrediction];

                        reason = `Following active pattern: ${activePatterns[0].pattern.join(' -> ')}`;

                        confidence = "High Confidence";

                    }

                }

                return { type: 'play', recommendations: finalRecommendations, reason, confidence };

            }

            function getStats(log = confirmedWinsLog) {

                const stats = {};

                allPredictionTypes.forEach(type => {

                    stats[type.id] = { hits: 0, total: 0, streak: 0, hitRate: 0, type: type };

                });


                log.forEach(item => {

                    if (item.status === 'success' && item.successfulPrediction && stats[item.successfulPrediction.id]) {

                        stats[item.successfulPrediction.id].hits++;

                    }

                    item.predictions.forEach(p => {

                        if(stats[p.id]) stats[p.id].total++;

                    });

                });

                

                for (const key in stats) {

                    if (stats[key].total > 0) {

                        stats[key].hitRate = stats[key].hits / stats[key].total;

                    }

                }


                for (const key in stats) {

                    let currentStreak = 0;

                    for (const item of log) {

                        if (item.successfulPrediction && item.successfulPrediction.id === key) {

                            currentStreak++;

                        } else {

                            break;

                        }

                    }

                    stats[key].streak = currentStreak;

                }

                return stats;

            }

            function getNeighbors(number, distance = 1) {

                const index = rouletteWheel.indexOf(number);

                if (index === -1) return [];

                const neighbors = [];

                for(let i = 1; i <= distance; i++) {

                    neighbors.push(rouletteWheel[(index - i + rouletteWheel.length) % rouletteWheel.length]);

                    neighbors.push(rouletteWheel[(index + i) % rouletteWheel.length]);

                }

                return [...new Set(neighbors)];

            }

            

            function getShortestPocketDistance(number, hitZone) {

                if(!hitZone || hitZone.length === 0 || number === null) return -1;

                const numIndex = rouletteWheel.indexOf(number);

                if(numIndex === -1) return -1;


                let minDistance = Infinity;

                hitZone.forEach(zoneNum => {

                    const zoneIndex = rouletteWheel.indexOf(zoneNum);

                    if(zoneIndex !== -1) {

                        const diff = Math.abs(numIndex - zoneIndex);

                        const distance = Math.min(diff, rouletteWheel.length - diff);

                        if(distance < minDistance) {

                            minDistance = distance;

                        }

                    }

                });

                return minDistance === Infinity ? -1 : minDistance;

            }

            function calculateVolatility(log = confirmedWinsLog) {

                const last10 = log.slice(0, 10);

                if (last10.length < 5) return 'Normal';

                

                const successCount = last10.filter(item => item.status === 'success').length;

                const predictability = successCount / last10.length;


                if (predictability > 0.7) return 'Stable';

                if (predictability < 0.3) return 'Choppy';

                return 'Normal';

            }


            function calculateCorrelationMatrix(log = confirmedWinsLog) {

                const matrix = {};

                const wins = log.filter(i => i.successfulPrediction).reverse();

                if (wins.length < 2) return {};


                for (let i = 0; i < wins.length - 1; i++) {

                    const currentGroup = wins[i].successfulPrediction.id;

                    const nextGroup = wins[i + 1].successfulPrediction.id;

                    if (!matrix[currentGroup]) matrix[currentGroup] = {};

                    if (!matrix[currentGroup][nextGroup]) matrix[currentGroup][nextGroup] = 0;

                    matrix[currentGroup][nextGroup]++;

                }

                for(const fromGroup in matrix) {

                    const total = Object.values(matrix[fromGroup]).reduce((a, b) => a + b, 0);

                    for(const toGroup in matrix[fromGroup]) {

                        matrix[fromGroup][toGroup] /= total;

                    }

                }

                return matrix;

            }


            function findActivePatterns(log = confirmedWinsLog, minLength = 3, lookback = 30) {

                const wins = log.slice(0, lookback).filter(i => i.successfulPrediction).map(i => i.successfulPrediction.id).reverse();

                if (wins.length < minLength) return [];

                

                const patterns = {};

                for (let i = 0; i <= wins.length - minLength; i++) {

                    const sequence = wins.slice(i, i + minLength);

                    const key = sequence.join(',');

                    patterns[key] = (patterns[key] || 0) + 1;

                }


                const lastSequence = wins.slice(wins.length - (minLength - 1));

                const active = [];

                for(const p in patterns) {

                    if(patterns[p] > 1) { 

                        const sequence = p.split(',');

                        const patternStart = sequence.slice(0, minLength - 1);

                        if(JSON.stringify(patternStart) === JSON.stringify(lastSequence)) {

                            active.push({ pattern: sequence, next: sequence[minLength - 1] });

                        }

                    }

                }

                return active;

            }

            function buildFeatureVector(currentPredictions, historySlice) {

                const vector = [];

                const lastWin = historySlice.length > 0 ? historySlice[0] : null;

                

                currentPredictions.forEach(p => vector.push(p.base / 36.0, p.terminal / 36.0));


                const pastStats = getStats(historySlice);

                allPredictionTypes.forEach(p_type => {

                    const stat = pastStats[p_type.id];

                    vector.push(stat.hitRate, stat.streak / 10.0);

                });


                const volatility = calculateVolatility(historySlice);

                vector.push(volatility === 'Stable' ? 1:0, volatility === 'Normal' ? 1:0, volatility === 'Choppy' ? 1:0);


                if(lastWin) {

                    currentPredictions.forEach(p => {

                        vector.push(getShortestPocketDistance(lastWin.winningNumber, p.hitZone) / 18.0);

                    });

                } else {

                    currentPredictions.forEach(() => vector.push(0));

                }

                

                return vector;

            }


            async function trainModel() {

                if (isTraining) return;

                const minData = 50;

                if (confirmedWinsLog.length < minData) {

                    alert(`Not enough data. Need at least ${minData} confirmed spins to train the model.`);

                    return;

                }


                isTraining = true;

                dom.trainModelButton.disabled = true;

                dom.modelStatus.textContent = 'Training... Please wait.';


                try {

                    const features = [];

                    const labels = [];

                    

                    for (let i = 0; i < confirmedWinsLog.length - 1; i++) {

                        const currentSpin = confirmedWinsLog[i];

                        if(!currentSpin.successfulPrediction) continue;


                        const pastHistory = confirmedWinsLog.slice(i + 1);

                        const featureVector = buildFeatureVector(currentSpin.predictions, pastHistory);

                        features.push(featureVector);


                        const labelIndex = allPredictionTypes.findIndex(p => p.id === currentSpin.successfulPrediction.id);

                        labels.push(labelIndex);

                    }

                    

                    if (features.length === 0) throw new Error("Not enough sequential data to create features.");


                    numFeatures = features[0].length;

                    const numClasses = allPredictionTypes.length;


                    const xs = tf.tensor2d(features);

                    const ys = tf.oneHot(tf.tensor1d(labels, 'int32'), numClasses);


                    const model = tf.sequential();

                    model.add(tf.layers.dense({ units: 128, activation: 'relu', inputShape: [numFeatures] }));

                    model.add(tf.layers.dropout({rate: 0.3}));

                    model.add(tf.layers.dense({ units: 64, activation: 'relu' }));

                    model.add(tf.layers.dropout({rate: 0.3}));

                    model.add(tf.layers.dense({ units: numClasses, activation: 'softmax' }));


                    model.compile({ optimizer: tf.train.adam(0.001), loss: 'categoricalCrossentropy', metrics: ['accuracy'] });


                    const historyData = await model.fit(xs, ys, {

                        epochs: 75,

                        validationSplit: 0.2,

                        callbacks: {

                            onEpochEnd: (epoch, logs) => {

                                dom.modelStatus.textContent = `Training... Epoch ${epoch+1}/75, Acc: ${logs.acc.toFixed(2)}`;

                            }

                        }

                    });


                    trainedModel = model;

                    const finalAccuracy = (historyData.history.acc.pop() * 100).toFixed(1);

                    dom.modelStatus.textContent = `Model trained! Accuracy: ${finalAccuracy}%`;

                    dom.contextualAIToggle.disabled = false;


                } catch (error) {

                    console.error("Model training failed:", error);

                    dom.modelStatus.textContent = 'Model training failed. See console.';

                } finally {

                    isTraining = false;

                    dom.trainModelButton.disabled = false;

                }

            }


            function getAIPrediction(currentPredictions) {

                if (!trainedModel) return null;

                

                const featureVector = buildFeatureVector(currentPredictions, confirmedWinsLog);

                const input = tf.tensor2d([featureVector]);

                const prediction = trainedModel.predict(input);

                const predictedIndex = prediction.as1D().argMax().dataSync()[0];

                

                return allPredictionTypes[predictedIndex].id;

            }

            function analyzeHistoricalData() {

                const text = dom.historicalNumbersInput.value;

                if (!text.trim()) {

                    dom.historicalAnalysisMessage.textContent = "Please paste some numbers first.";

                    return;

                }

                const numbers = text.match(/\d+/g).map(Number).reverse();

                if (numbers.length < 2) {

                    dom.historicalAnalysisMessage.textContent = "Need at least two numbers to create a history entry.";

                    return;

                }


                history = [];

                confirmedWinsLog = [];


                for (let i = 0; i < numbers.length - 1; i++) {

                    const num1 = numbers[i];

                    const num2 = numbers[i+1];

                    const winningNumber = numbers[i+2];

                    

                    if(winningNumber === undefined) break;


                    const predictions = calculatePredictions(num1, num2);

                    const successfulPrediction = predictions.find(p => p.hitZone.includes(winningNumber));

                    

                    const newItem = {

                        id: Date.now() + i,

                        num1, num2, predictions,

                        winningNumber: winningNumber,

                        status: successfulPrediction ? 'success' : 'fail',

                        successfulPrediction: successfulPrediction || null,

                        recommendation: null

                    };


                    history.unshift(newItem);

                    confirmedWinsLog.unshift(newItem);

                }

                dom.historicalAnalysisMessage.textContent = `Successfully processed ${history.length} historical entries.`;

                renderAll();

                saveState();

            }


            async function processImage(file) {

                if (!file) return;

                if (GOOGLE_API_KEY === "PASTE_YOUR_NEW_SECRET_KEY_HERE") {

                    dom.ocrStatus.textContent = "Error: Google API Key is not set.";

                    return;

                }

                dom.ocrStatus.textContent = "Reading image...";

                dom.analyzeHistoricalDataButton.disabled = true;


                const reader = new FileReader();

                reader.onload = async (e) => {

                    const base64Image = e.target.result.split(',')[1];

                    const requestBody = {

                        requests: [{

                            image: { content: base64Image },

                            features: [{ type: 'TEXT_DETECTION' }]

                        }]

                    };

                    try {

                        const response = await fetch(`https://vision.googleapis.com/v1/images:annotate?key=${GOOGLE_API_KEY}`, {

                            method: 'POST',

                            headers: { 'Content-Type': 'application/json' },

                            body: JSON.stringify(requestBody)

                        });

                        if (!response.ok) {

                            throw new Error(`API request failed: ${response.statusText}`);

                        }

                        const data = await response.json();

                        const text = data.responses[0]?.fullTextAnnotation?.text;

                        if (text) {

                            dom.historicalNumbersInput.value = text;

                            dom.ocrStatus.textContent = "Image text extracted!";

                        } else {

                            dom.ocrStatus.textContent = "No text found in image.";

                        }

                    } catch (error) {

                        console.error("OCR Error:", error);

                        dom.ocrStatus.textContent = "OCR failed. See console for details.";

                    } finally {

                        dom.analyzeHistoricalDataButton.disabled = false;

                    }

                };

                reader.readAsDataURL(file);

            }

            function drawTrackingSimulation(timestamp) {

                const canvas = dom.trackingCanvas;

                const ctx = canvas.getContext('2d');

                canvas.width = canvas.clientWidth;

                canvas.height = canvas.clientHeight;

                

                ctx.clearRect(0, 0, canvas.width, canvas.height);


                // Simulate wheel tracking

                const wheelRadius = Math.min(canvas.width, canvas.height) * 0.4;

                const centerX = canvas.width / 2;

                const centerY = canvas.height / 2;

                ctx.strokeStyle = 'rgba(255, 255, 0, 0.7)';

                ctx.lineWidth = 3;

                ctx.beginPath();

                ctx.arc(centerX, centerY, wheelRadius, 0, Math.PI * 2);

                ctx.stroke();


                // Simulate ball tracking

                const ballOrbitRadius = wheelRadius * (0.8 - (timestamp % 5000) / 10000); // Decaying orbit

                const ballAngle = (timestamp / 500) % (Math.PI * 2);

                const ballX = centerX + ballOrbitRadius * Math.cos(ballAngle);

                const ballY = centerY + ballOrbitRadius * Math.sin(ballAngle);

                ctx.fillStyle = 'rgba(0, 255, 255, 0.9)';

                ctx.beginPath();

                ctx.arc(ballX, ballY, 8, 0, Math.PI * 2);

                ctx.fill();

                

                // Simulate prediction

                if (timestamp % 7000 > 6500) { // Display prediction for a short while every 7 seconds

                    const predictedSector = Math.floor(Math.random() * 37);

                    const neighbors = getNeighbors(predictedSector, 2);

                    dom.physicsPrediction.textContent = `Prediction: ${[predictedSector, ...neighbors].join(', ')}`;

                } else if (timestamp % 7000 < 100) {

                    dom.physicsPrediction.textContent = ''; // Clear prediction

                }

                

                visionAnimationId = requestAnimationFrame(drawTrackingSimulation);

            }
